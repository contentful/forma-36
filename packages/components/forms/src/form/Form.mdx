`Form` is a wrapper component to apply a `<form />` wrapper and handle form submissions.

## Table of contents

- [How to use Form](#how-to-use-Form)

## How to use Form

- The `Form` component should wrap form elements into `<form />`
- Provide an `onSubmit` handler for form validation and submission
- state of the validation of the form should be maniputated from outside

### Basic example

```jsx
function FormaExample() {
  const [submited, setSubmited] = React.useState(false);
  return (
    <Form onSubmit={() => setSubmited(true)}>
      <FormControl>
        <FormControl.Label isRequired>Name</FormControl.Label>
        <TextInput />
        <FormControl.HelpText>
          Please enter your first name
        </FormControl.HelpText>
      </FormControl>

      <FormControl>
        <FormControl.Label>Description</FormControl.Label>
        <Box>
          <Textarea />
        </Box>
        <FormControl.HelpText>Tell me about youself</FormControl.HelpText>
      </FormControl>
      <Button variant={'primary'} type="submit" isDisabled={submited}>
        {submited ? 'Sumbited' : 'Click me to submit'}
      </Button>
    </Form>
  );
}
```

### Example with counting characters

```jsx
function FormaExample() {
  const [submited, setSubmited] = React.useState(false);
  return (
    <Form onSubmit={() => setSubmited(true)}>
      <FormControl>
        <FormControl.Label isRequired>user name</FormControl.Label>
        <TextInput />
        <FormControl.HelpText>Please enter your username</FormControl.HelpText>
      </FormControl>
      <FormControl {...args}>
        <FormControl.Label isRequired>password</FormControl.Label>
        <TextInput maxLength={10} />
        <Flex justifyContent="space-between">
          <FormControl.HelpText>
            Password can only have 10 characters
          </FormControl.HelpText>
          <FormControl.Counter />
        </Flex>
      </FormControl>
      <Button variant={'primary'} type="submit" isDisabled={submited}>
        {submited ? 'Sumbited' : 'Click me to submit'}
      </Button>
    </Form>
  );
}
```

### Example using solution like Formik

```jsx static=true
import { Formik, Field } from 'formik';
import { Form, FormControl, TextInput } from '@contentful/f36-forms';
import { Button } from '@contentful/f36-button';

function ExampleUsingFormik() {
  function validateName(value) {
    let error;
    if (!value) {
      error = 'Name is required';
    }
    return error;
  }

  return (
    <Formik
      onSubmit={(values, actions) => {
        setTimeout(() => {
          alert(JSON.stringify(values, null, 2));
          actions.setSubmitting(false);
        }, 1000);
      }}
    >
      {(props) => (
        <Form>
          <Field name="name" validate={validateName}>
            {({ field, form }) => (
              <FormControl isInvalid={form.errors.name}>
                <FormControl.Label>First name</FormControl.Label>
                <TextInput {...field} id="name" placeholder="name" />
                {form.errors.name && (
                  <FormControl.ValidationMessage>
                    {form.errors.name}
                  </FormControl.ValidationMessage>
                )}
              </FormControl>
            )}
          </Field>
          <Button variant="primary" type="submit">
            Sumbit
          </Button>
        </Form>
      )}
    </Formik>
  );
}
```
