  const [isShown, setShown] = React.useState(false);

              </Button>
            </Modal.Controls>
          </React.Fragment>

      </Modal>
    </React.Fragment>

}

## Best practices <a name="best-practices" />

- Use modal dialogs to request the user to enter information critical to continuing the current process.
- Use modal dialogs for important warnings, as a way to prevent or correct critical errors.
- Do not use modal dialogs for nonessential information that is not related to the current user flow.
- Avoid modal dialogs that interrupt high-stake processes such as checkout flows.

## Variations <a name="variations" />

The Modal component can be configured in a number of different ways. Here is a guide to when to use certain variations.

### Position (`position`) <a name="position" />

```jsx
function ModalExample() {
  const [isShown, setShown] = React.useState(false);
  const [position, setPosition] = React.useState('top');
  return (
    <React.Fragment>
      <Form>
        <FieldGroup row>
          <Button
            buttonType="positive"
            onClick={() => {
              setPosition('top');
              setShown(true);
            }}
          >
            Show top modal
          </Button>
          <Button
            onClick={() => {
              setPosition('center');
              setShown(true);
            }}
          >
            Show center modal
          </Button>
        </FieldGroup>
      </Form>
      <Modal title="Centered modal" isShown={isShown} position={position}>
        {() => (
          <React.Fragment>
            <Modal.Header title="Title" />
            <Modal.Content>Hello from controlled modal window</Modal.Content>
            <Modal.Controls>
              <Button buttonType="positive" onClick={() => setShown(false)}>
                Confirm
              </Button>
              <Button buttonType="muted" onClick={() => setShown(false)}>
                Close
              </Button>
            </Modal.Controls>
          </React.Fragment>
        )}
      </Modal>
    </React.Fragment>
  );
}
```

### Sizes (`size`)

```jsx
function ModalExample() {
  const [isShown, setShown] = React.useState(false);
  const [size, setSize] = React.useState('medium');
  return (
    <React.Fragment>
      <Form>
        <SelectField
          onChange={e => {
            setSize(e.target.value);
          }}
          labelText="Select a width"
        >
          <Option value="small">small</Option>
          <Option value="medium">medium</Option>
          <Option value="large">large</Option>
          <Option value="fullWidth">full width</Option>
          <Option value="200px">200px (specific width value)</Option>
        </SelectField>
        <Button
          onClick={() => {
            setShown(true);
          }}
        >
          Show modal
        </Button>
      </Form>
      <Modal title="Centered modal" isShown={isShown} size={size}>
        {() => (
          <React.Fragment>
            <Modal.Header title="Title" />
            <Modal.Content>Hello from controlled modal window</Modal.Content>
            <Modal.Controls>
              <Button buttonType="positive" onClick={() => setShown(false)}>
                Confirm
              </Button>
              <Button buttonType="muted" onClick={() => setShown(false)}>
                Close
              </Button>
            </Modal.Controls>
          </React.Fragment>
        )}
      </Modal>
    </React.Fragment>
  );
}
```
