## Examples

### Example title (variant name)

Optional: Short "when to use" with considerations towards the end user

```jsx
() => {
  const menuListRef = React.useRef(null);

  const {
    focusedIndex,
    handleArrowsKeyDown,
    setFocusedIndex,
  } = useArrowKeyNavigation({
    itemsContainerRef: menuListRef,
    itemsSelector: '[class="menu-button"]',
    keyType: 'vertical',
  });

  useEffect(() => {
    if (menuListRef.current) {
      const menuItems = menuListRef.current.querySelectorAll(
        '[class="menu-button"]',
      );

      if (menuItems.length > 0 && focusedIndex < menuItems.length) {
        menuItems[focusedIndex].focus({
          preventScroll: false,
        });
      }
    }
  }, [focusedIndex]);

  <Stack ref={menuListRef}>
    <Button className="menu-button">First Button</Button>
    <Button className="menu-button">Second Button</Button>
    <Button className="menu-button">Third Button</Button>
  </Stack>;
};
```
